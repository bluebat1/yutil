- 时间

```go

```

- [示例]TimeInterval

```go
func TestTimeInterval(t *testing.T) {
	sum := 0
	TimeInterval()
	for i := 0; i < 1000000000; i ++ {
		sum += 1
	}
	TimeInterval()
	for i := 0; i < 1000000000; i ++ {
		sum += 1
	}
	TimeInterval()
	TimeInterval(0)                       // 重置计时
	for i := 0; i < 1000000000; i ++ {
		sum += 1
	}
	TimeInterval()
	// out
	// 2019/12/17 11:08:43 1.cost: 0.329259s  // 第一次耗时
	// 2019/12/17 11:08:43 2.cost: 0.670507s  // 第二次耗时
	// 2019/12/17 11:08:44 1.cost: 0.329759s  // 第三次耗时(计时器被重置)
}
```
